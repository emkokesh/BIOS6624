#apply 10-year censoring (365.25 x 10 = 3652.5 days)
days_10yr <- 365.25 * 10

analytic <- analytic %>%
  mutate(
    STROKE_10yr   = ifelse(STROKE == 1 & TIMESTRK <= days_10yr, 1, 0),
    TIMESTRK_10yr = ifelse(TIMESTRK > days_10yr, days_10yr, TIMESTRK)
  )

#number of strokes after 10 years that were re-coded as censored
sum(analytic$STROKE == 1 & analytic$TIMESTRK > days_10yr)

#flag strokes at time = 0- check there are no prevalent stroke cases remaining after exclusion

  AGE     = list(lo = 32,  hi = 81),

map_dfr(names(cont_checks), function(v) {
  tibble(
    Variable   = v,
    Low_bound  = cont_checks[[v]]$lo,
    N_below    = sum(analytic[[v]] < cont_checks[[v]]$lo, na.rm = TRUE),
    High_bound = cont_checks[[v]]$hi,
    N_above    = sum(analytic[[v]] > cont_checks[[v]]$hi, na.rm = TRUE)
  )
})

#do bivariate analysis by sex for binary variables
bin_results <- analytic %>%
  mutate(sex = ifelse(SEX == 1, "men", "women")) %>%
  group_split(sex) %>%
  map_dfr(function(df_sex) {
    
    sex_label <- unique(df_sex$sex)
    
    map_dfr(bin_vars, function(v) {
      df_use <- df_sex[!is.na(df_sex[[v]]), ]
      tbl <- table(df_use[[v]], df_use$STROKE_10yr)
      
      test <- if (any(tbl < 5)) {
        fisher.test(tbl)
      } else {
        suppressWarnings(chisq.test(tbl))
      }
      
      stroke_pct_no <- df_use %>%
        filter(.data[[v]] == 0) %>%
        summarise(stroke_pct = round(mean(STROKE_10yr) * 100, 1)) %>%
        pull(stroke_pct)
      
      stroke_pct_yes <- df_use %>%
        filter(.data[[v]] == 1) %>%
        summarise(stroke_pct = round(mean(STROKE_10yr) * 100, 1)) %>%
        pull(stroke_pct)
      
      tibble(
        sex = sex_label,
        variable = v,
        stroke_pct_no = stroke_pct_no,
        stroke_pct_yes = stroke_pct_yes,
        p_value = round(test$p.value, 4),
        test_used = if (any(tbl < 5)) "fisher" else "chi-square"
      )
    })
  })
